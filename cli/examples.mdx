This page provides practical examples of using the MCPJam Inspector CLI in various development scenarios and workflows.

## Quick Testing Examples

### Test a Python MCP Server

```bash
# Simple FastMCP server
npx @mcpjam/inspector@latest uv run fastmcp run /Users/dev/weather-server/main.py

# Traditional Python server with virtual environment
npx @mcpjam/inspector@latest ./venv/bin/python server.py

# Python server with configuration
npx @mcpjam/inspector@latest python server.py --config config.json --debug
```

### Test a Node.js MCP Server

```bash
# Direct Node.js execution
npx @mcpjam/inspector@latest node /Users/dev/mcp-server/index.js

# NPX package execution
npx @mcpjam/inspector@latest npx -y @mycompany/mcp-tools-server

# TypeScript server with ts-node
npx @mcpjam/inspector@latest npx ts-node server.ts
```

### Test Compiled Servers

```bash
# Go binary
npx @mcpjam/inspector@latest ./dist/mcp-server-go

# Rust binary
npx @mcpjam/inspector@latest ./target/release/mcp-server

# C++ binary with arguments
npx @mcpjam/inspector@latest ./build/mcp-server --config config.toml
```

## Development Workflows

### Local Development Setup

```bash
# Development environment with debugging
npx @mcpjam/inspector@latest \
  --port 3001 \
  -e NODE_ENV=development \
  -e DEBUG=true \
  -e LOG_LEVEL=debug \
  python server.py
```

### Hot Reload Development

```bash
# Python with auto-reload (using watchdog)
npx @mcpjam/inspector@latest \
  -e PYTHONDONTWRITEBYTECODE=1 \
  -e FLASK_ENV=development \
  python -m flask run --host 0.0.0.0 --port 8080

# Node.js with nodemon
npx @mcpjam/inspector@latest nodemon --exec "node server.js"
```

### Multi-Environment Testing

```bash
# Development
npx @mcpjam/inspector@latest \
  --port 3001 \
  --env-file .env.development \
  python server.py

# Staging  
npx @mcpjam/inspector@latest \
  --port 3002 \
  --env-file .env.staging \
  python server.py

# Production (local testing)
npx @mcpjam/inspector@latest \
  --port 3003 \
  --env-file .env.production \
  python server.py
```

## Testing and CI/CD

### Automated Testing

```bash
# Test runner integration
npx @mcpjam/inspector@latest \
  --port 3001 \
  -e TEST_MODE=true \
  -e MOCK_EXTERNAL_APIS=true \
  python test_server.py &

# Run tests against the server
sleep 5  # Wait for server startup
python -m pytest tests/
kill $!  # Stop inspector
```

### Docker Integration

```bash
# Test against dockerized dependencies
docker run -d --name postgres -p 5432:5432 -e POSTGRES_PASSWORD=test postgres:13
docker run -d --name redis -p 6379:6379 redis:alpine

npx @mcpjam/inspector@latest \
  -e DATABASE_URL=postgresql://postgres:test@localhost:5432/postgres \
  -e REDIS_URL=redis://localhost:6379 \
  python docker_integrated_server.py

# Cleanup
docker stop postgres redis
docker rm postgres redis
```

### Performance Testing

```bash
# Performance testing setup
npx @mcpjam/inspector@latest \
  --port 3001 \
  -e PERFORMANCE_MODE=true \
  -e DISABLE_DEBUG_LOGS=true \
  -e CACHE_ENABLED=true \
  python high_performance_server.py &

# Run load tests
ab -n 1000 -c 10 http://localhost:3001/api/health
kill $!
```

## Debugging Examples

### Verbose Logging

```bash
# Full debug output
DEBUG=* npx @mcpjam/inspector@latest python server.py

# Specific debug categories
DEBUG=mcpjam:server,mcpjam:mcp npx @mcpjam/inspector@latest python server.py

# Redirect debug output to file
DEBUG=* npx @mcpjam/inspector@latest python server.py 2>&1 | tee debug.log
```

### Server Troubleshooting

```bash

python server.py  # Should work standalone

# Check for port conflicts

```

### Network Debugging

### Load Balancer Testing

```bash
# Multiple server instances
npx @mcpjam/inspector@latest --port 3001 python server.py &
npx @mcpjam/inspector@latest --port 3002 python server.py &
npx @mcpjam/inspector@latest --port 3003 python server.py &

# Test load balancing
for i in {1..100}; do
  curl http://localhost:300$((1 + RANDOM % 3))/api/health
done

# Cleanup
jobs -p | xargs kill
```

## Scripting and Automation

### Bash Script Integration

```bash
#!/bin/bash
# start-mcp-server.sh

set -e

# Check prerequisites
command -v python3 >/dev/null 2>&1 || { echo "Python 3 required" >&2; exit 1; }
command -v npx >/dev/null 2>&1 || { echo "Node.js/NPX required" >&2; exit 1; }

# Load environment
if [ -f .env ]; then
  set -a
  source .env
  set +a
fi

# Start server
echo "Starting MCP Inspector with server..."
exec npx @mcpjam/inspector@latest python server.py "$@"
```

### Make Integration

```makefile
# Makefile
.PHONY: dev test prod

dev:
	DEBUG=true npx @mcpjam/inspector@latest python server.py

test:
	TEST_MODE=true npx @mcpjam/inspector@latest --port 3001 python server.py &
	sleep 5
	python -m pytest tests/
	pkill -f inspector || true

prod:
	NODE_ENV=production npx @mcpjam/inspector@latest \
		--env-file .env.production \
		python server.py
```
